import { Unstyled } from "@storybook/blocks";
import { Section } from '@phoenix-ui/section'
import { variants } from '@phoenix-ui/baseline/tailwind/attributes/margin'

# Demo of supported padding values
Tailwind supports values up to padding(96), initially phoenix-ui will go as far as 40.

## Example
```
<Section margin={{ all: '0' }}Content</Section>
```
## Types
[See Margin Values](https://github.com/bolanosdev/phoenix-ui/blob/main/libraries/types/spacing/padding.types.ts)
