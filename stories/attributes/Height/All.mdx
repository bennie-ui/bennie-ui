# Demo of supported margin values

## Example
```
<Section width={{ height: '0' }}Content</Section>
```
## Types
[See Margin Values](https://github.com/bolanosdev/bennie-ui/blob/main/libraries/types/sizing/width.types.ts)


# Compiled tests
{
  variants.base.map((value, idx) => {
    const full_clases = ["h-full","h-screen","h-svw","h-lvw","h-min","h-max","h-fit"]
    const h_value = full_clases.includes(value) ? value: value.split('-')[1]

    return <Unstyled>
       <Section height={{value: 96}} className="border" margin={{bottom: '5'}}>
        <Section
          height={{ value: h_value }} className="border border-sky-500">
          {value}
        </Section>
       </Section>
    </Unstyled>
  })
}
