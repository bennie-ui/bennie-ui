import { Unstyled } from "@storybook/blocks";
import { Section } from '@phoenix-ui/section'
import { variants } from '@phoenix-ui/baseline/tailwind/attributes/width'

# Demo of supported margin values

## Example
```
<Section width={{ value: '0' }}Content</Section>
```
## Types
[See Margin Values](https://github.com/bolanosdev/phoenix-ui/blob/main/libraries/types/sizing/width.types.ts)


# Compiled tests
{
  variants.base.map((value, idx) => {
    const full_clases = ["w-full","w-screen","w-svw","w-lvw","w-min","w-max","w-fit"]
    const w_value = full_clases.includes(value) ? value: value.split('-')[1]

    return <Unstyled key={idx}>
      <Section width={{value: 'full'}} className="border">
        <Section width={{ value: w_value }} className="border border-sky-500">
          {value}
        </Section>
      </Section>
    </Unstyled>
  })
}
