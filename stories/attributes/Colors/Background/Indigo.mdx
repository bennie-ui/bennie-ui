import { Unstyled } from "@storybook/blocks";
import { Section } from '@bennie-ui/section'
import { Text } from '@bennie-ui/text'
import { variants } from '@bennie-ui/baseline/tailwind/colors/background/sm'

# Demo of supported text-indigo-* values

## Example
```
<Text colors={{ background: { color: 'indigo', weight: 400 } }}>Content</Text>
```
## Types
[See Margin Values](https://github.com/bolanosdev/bennie-ui/blob/main/libraries/types/sizing/width.types.ts)


# Indigo tests
{
  variants.base.indigo.map((value, idx) => {
    const [_, color, weight] =  value.split('-')
    return <Unstyled key={idx}>
      <Section className="w-full" colors={{ background: {color, weight}, text: { color: 'slate' } }}>
          {value}
      </Section>
    </Unstyled>
  })
}
