import { Unstyled } from "@storybook/blocks";
import { Section } from '@phoenix-ui/section'
import { Text } from '@phoenix-ui/text'
import { variants } from '@phoenix-ui/baseline/tailwind/colors/text/sm'

# Demo of supported text-cyan-* values

## Example
```
<Text colors={{ text: { color: 'cyan', weight: 400 } }}>Content</Text>
```
## Types
[See Margin Values](https://github.com/bolanosdev/phoenix-ui/blob/main/libraries/types/sizing/width.types.ts)


# Neutral tests
{
  variants.base.cyan.map((value, idx) => {
    const [_, color, weight] =  value.split('-')

    return <Unstyled key={idx}>
      <Section className="w-full">
        <Text colors={{ text: {color, weight}, background: { color: 'white', weight: 100 } }}>
          {value}
       </Text> 
      </Section>
    </Unstyled>
  })
}
