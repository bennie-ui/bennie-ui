import { Unstyled } from "@storybook/blocks";
import { Section } from '@bennie-ui/section'
import { Text } from '@bennie-ui/text'
import { variants } from '@bennie-ui/baseline/tailwind/colors/text/sm'

# Demo of supported text-yellow-* values

## Example
```
<Text colors={{ text: { color: 'yellow', weight: 400 } }}>Content</Text>
```
## Types
[See Margin Values](https://github.com/bolanosdev/bennie-ui/blob/main/libraries/types/sizing/width.types.ts)


# Neutral tests
{
  variants.base.yellow.map((value, idx) => {
    const [_, color, weight] =  value.split('-')

    return <Unstyled key={idx}>
      <Section className="w-full">
        <Text colors={{ text: {color, weight}, background: { color: 'white', weight: 100 } }}>
          {value}
       </Text> 
      </Section>
    </Unstyled>
  })
}
